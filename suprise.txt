        function fadeInMessage(id) {
            const message = document.getElementById(id);
            setTimeout(() => {
                message.style.opacity = 1;
            }, 1000);
        }

        function fadeOutMessage(id) {
            const message = document.getElementById(id);
            message.style.opacity = 0;
        }

        function startGame() {
            fadeInMessage("message1");
            
            setTimeout(() => {
                fadeOutMessage("message1");
                fadeInMessage("message2");
            }, 4000);
            
            setTimeout(() => {
                fadeOutMessage("message2");
                fadeInMessage("message3");
                setTimeout(() => {
                    fadeOutMessage("message3");
                    displayQuestion();
                }, 4000);
            }, 8000);
        }

        function displayQuestion() {
            const questionData = questions[currentQuestionIndex];
            document.getElementById("question").textContent = questionData.question;
            document.getElementById("option1").textContent = questionData.options[0];
            document.getElementById("option2").textContent = questionData.options[1];
            document.getElementById("option3").textContent = questionData.options[2];
            document.getElementById("option4").textContent = questionData.options[3];

            if (currentQuestionIndex === 1 || currentQuestionIndex === 2) {
                document.getElementById("option4").style.display = 'none';
            } else {
                document.getElementById("option4").style.display = 'block';
            }

            document.getElementById("question").style.opacity = 1;
            document.getElementById("option1").style.opacity = 1;
            document.getElementById("option2").style.opacity = 1;
            document.getElementById("option3").style.opacity = 1;
        }

        function handleAnswer(answerIndex) {
            const questionData = questions[currentQuestionIndex];
            const imageElement = document.getElementById("image");
            const imageCaptionElement = document.getElementById("imageCaption");
            const nextButton = document.getElementById("nextButton");

            if (answerIndex === questionData.correctAnswer) {
                imageElement.innerHTML = `<img src="${questionData.image}" alt="Surprise">`;
                imageCaptionElement.textContent = questionData.caption; 
                imageElement.style.display = 'block';
                imageElement.style.animation = 'flyIn 1s forwards';
                imageCaptionElement.style.display = 'block';
                nextButton.style.display = 'block'; 
            } else {
                alert("Try again!");
            }
        }

        function nextQuestion() {
            const nextButton = document.getElementById("nextButton");
            nextButton.style.display = 'none'; 
            const imageElement = document.getElementById("image");
            const imageCaptionElement = document.getElementById("imageCaption");

            imageElement.style.display = 'none'; 
            imageCaptionElement.style.display = 'none';

            if (currentQuestionIndex < questions.length - 1) {
                currentQuestionIndex++;
                displayQuestion();
            } else {
                showFinalSummary();
            }
        }

        function showFinalSummary() {
            document.getElementById("finalMusic").play();
            const finalSummary = document.getElementById("finalSummary");
            finalSummary.style.display = 'block';
            finalSummary.style.opacity = 1;
        }

        function goToFinalPage() {
            document.getElementById("finalSummary").style.display = 'none';
            document.getElementById("thankYouPage").style.display = 'block'; 
        }

        function downloadImage(imageName) {
            const link = document.createElement("a");
            link.href = imageName; 
            link.download = imageName;
            link.click();
        }
